import numpy as np import math import matplotlib.pyplot as plt from mpl_toolkits.mplot3d import Axes3D # Prompt for beam length L = float(input("Beam length (ft): ")) # Prompt for point loads point_loads = [] while True: inp = input("Location (ft) and Pu (kips) separated by space, or empty to stop: ") if not inp: break loc, pu = map(float, inp.split()) if loc > L or loc < 0: print("Location must be between 0 and beam length. Skipping.") continue point_loads.append((loc, pu)) # Prompt for ratio ratio = float(input("Ratio of combined ultimate load to DL: ")) # Prompt for max deflection delta_max = float(input("Max deflection (in): ")) # Calculate Vu and Mu if point_loads: R_left = sum(pu * (L - loc) / L for loc, pu in point_loads) R_right = sum(pu * loc / L for loc, pu in point_loads) Vu = max(R_left, R_right) # For Mu num_points = 1000 x = np.linspace(0, L, num_points) dx = x[1] - x[0] V_arr = np.full(num_points, R_left) for loc, pu in point_loads: V_arr[x > loc] -= pu M_arr = np.cumsum(V_arr) * dx Mu = max(abs(M_arr)) else: Vu = 0.0 Mu = 0.0 # For EI required point_loads_dl = [(loc, pu / ratio) for loc, pu in point_loads] L_in = L * 12 x_in = np.linspace(0, L_in, 1000) def deflection_at_x(x, a, P, L, EI): b = L - a if x <= a: return P * b * x * (L**2 - b**2 - x**2) / (6 * EI * L) else: return P * a * (L - x) * (L**2 - a**2 - (L - x)**2) / (6 * EI * L) if point_loads_dl: v = np.zeros_like(x_in) for loc, p_dl in point_loads_dl: a = loc * 12 P = p_dl * 1000 # lbs for i, xi in enumerate(x_in): v[i] += deflection_at_x(xi, a, P, L_in, EI=1) max_def_EI1 = max(v) EI_req = max_def_EI1 / delta_max if delta_max != 0 else 0.0 else: EI_req = 0.0 # Printout of the beam beam_visual_scale = max(int(L * 2), 1) print("Simply Supported Beam:") print("A" + "─" * beam_visual_scale + "B") for loc, pu in point_loads: pos = int(loc / L * beam_visual_scale) print(" " * pos + f"↓ Pu = {pu:.2f} kips") # Results print(f"\nMu = {Mu:.2f} kip-ft") print(f"Vu = {Vu:.2f} kips") print(f"EI required = {EI_req:.2e} lb-in²") # Scientific notation for large numbers # Graphical 3D visualization fig = plt.figure() ax = fig.add_subplot(111, projection='3d') # Draw beam ax.plot([0, L], [0, 0], [0, 0], color='blue', linewidth=5, label='Beam') # Draw supports ax.plot([0, 0], [-0.5, 0.5], [0, 0], color='red', linewidth=3) ax.plot([L, L], [-0.5, 0.5], [0, 0], color='red', linewidth=3) # Draw point loads as arrows max_pu = max([pu for _, pu in point_loads] or [1]) for loc, pu in point_loads: arrow_length = pu / max_pu * 2 # Scale arrow length ax.quiver(loc, 0, arrow_length, 0, 0, -arrow_length, color='green', arrow_length_ratio=0.2) ax.set_xlim([0, L]) ax.set_ylim([-1, 1]) ax.set_zlim([-2, 2]) ax.set_xlabel('Length (ft)') ax.set_ylabel('Y') ax.set_zlabel('Z') ax.set_title('3D Beam Visualization') plt.show() # Deflection calculation saved for later (not printed)
